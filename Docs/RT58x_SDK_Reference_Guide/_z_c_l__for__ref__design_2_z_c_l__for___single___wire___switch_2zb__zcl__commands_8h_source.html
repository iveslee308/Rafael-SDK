621  zb_uint8_t resp_size = 0xff; \
622  (read_attr_resp) = zb_buf_len(data_buf) >= ZB_ZCL_READ_ATTR_RESP_SIZE ? \
623  (zb_zcl_read_attr_res_t*)zb_buf_begin(data_buf) : NULL; \
624  \
625  if (read_attr_resp != NULL) \

627  resp_size = ZB_ZCL_READ_ATTR_RESP_SIZE; \
628  ZB_ZCL_HTOLE16_INPLACE(&(read_attr_resp)->attr_id); \

630  { \
631  resp_size += \

633  zb_zcl_get_attribute_size((read_attr_resp)->attr_type, (read_attr_resp)->attr_value); \
634  if (resp_size <= zb_buf_len(data_buf)) \
635  { \

638  } \
639  \
640  if (resp_size <= zb_buf_len(data_buf)) \
641  { \
642  (void)zb_buf_cut_left((data_buf), resp_size); \
643  } \
